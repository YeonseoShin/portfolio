/**
 * 세 점의 좌표를 받아 b를 꼭짓점으로 하는 각도를 계산 (도 단위)
 * 자소서에 기재한 삼각함수 계산 로직
 */
export function calcAngle(
  a: NormalizedCoord,
  b: NormalizedCoord,
  c: NormalizedCoord
): number {
  const v1 = [a[0] - b[0], a[1] - b[1]]
  const v2 = [c[0] - b[0], c[1] - b[1]]
  const dot  = v1[0] * v2[0] + v1[1] * v2[1]
  const mag1 = Math.sqrt(v1[0] ** 2 + v1[1] ** 2)
  const mag2 = Math.sqrt(v2[0] ** 2 + v2[1] ** 2)
  if (!mag1 || !mag2) return 0
  const cos = Math.min(1, Math.max(-1, dot / (mag1 * mag2)))
  return Math.round(Math.acos(cos) * 180 / Math.PI)
}
src/components/SkeletonOverlay.tsx 
export default function SkeletonOverlay({ joints, imageLayout }: Props) {
  const { offsetX, offsetY, width, height, containerW, containerH } = imageLayout

  /** 정규화 좌표 → SVG viewBox(0~100) 좌표 변환 */
  function toSVG(coord: NormalizedCoord): [number, number] {
    return [
      ((offsetX + coord[0] * width)  / containerW) * 100,
      ((offsetY + coord[1] * height) / containerH) * 100,
    ]
  }

  return (
    <svg viewBox="0 0 100 100" style={{ position: 'absolute', inset: 0 }}>
      {CONNECTIONS.map(([a, b]) => {
        const [x1,y1] = toSVG(joints[a]!)
        const [x2,y2] = toSVG(joints[b]!)
        return <line key={`${a}-${b}`} x1={x1} y1={y1} x2={x2} y2={y2}
                 stroke="#6c7fff" strokeWidth="0.7" />
      })}
    </svg>
  )
}

import pandas as pd
import numpy  as np

def calc_trust_score(df: pd.DataFrame) -> pd.DataFrame:
    """출연금 · 보험금지급 · 회수율 기반 가중 점수 계산"""
    max_ch  = df['출연금'].max()    or 1
    max_ins = df['보험금지급'].max() or 1

    df['ch_score']  = 30 * (1 - df['출연금'] / max_ch)
    df['ins_score'] = np.where(
        df['보험금지급'] == 0, 20,
        20 * (1 - df['보험금지급'].clip(0, max_ins) / max_ins)
    )
    df['rec_score'] = 30 * df['회수율']
    df['bonus']     = np.where(
        (df['회수율'] > 0.5) & (df['출연금'] < max_ch * 0.3), 10, 4
    )
    df['TrustScore'] = (
        df[['ch_score','ins_score','rec_score','bonus']]
        .sum(axis=1).clip(0, 100).round(1)
    )
    return df.sort_values('TrustScore', ascending=False)

let dragCard = null;

function initCard(card) {
  card.addEventListener('dragstart', () => {
    dragCard = card;
    setTimeout(() => card.classList.add('dragging'), 0);
  });
  card.addEventListener('dragend', () => {
    card.classList.remove('dragging');
    dragCard = null;
    updateCounts();
  });
}

document.querySelectorAll('.col').forEach(col => {
  col.addEventListener('dragover', e => {
    e.preventDefault();
    col.classList.add('drag-over');
  });
  col.addEventListener('drop', e => {
    e.preventDefault();
    col.classList.remove('drag-over');
    if (!dragCard) return;
    col.querySelector('.col-body').appendChild(dragCard);
    const done = col.dataset.status === 'done';
    dragCard.style.opacity = done ? '0.6' : '1';
  });
});

  const choices = [...wrong, word].sort(() => Math.random() - 0.5);

  choices.forEach((c, i) => {
    const btn = optionBtns[i];
    btn.textContent = c.ko;
    btn.dataset.correct = String(c === word);
    btn.className = 'option-btn';
  });
  startTimer();
}

function handleAnswer(btn) {
  clearInterval(timerInterval);
  const isCorrect = btn.dataset.correct === 'true';
  if (isCorrect) {
    combo++;
    score += 10 + (combo - 1) * 5; // 콤보 보너스

    combo = 0;

      ?.classList.add('correct');

}

// 카테고리별 지출 집계 및 과소비 감지
function analyzeSpending() {
  const transactions = loadTransactions();
  const budget = loadBudget();

  // 카테고리별 합산
  const byCategory = transactions.reduce((acc, tx) => {
    acc[tx.category] = (acc[tx.category] ?? 0) + tx.amount;
    return acc;
  }, {});

  // 과소비 경고 생성
  const warnings = Object.entries(byCategory)
    .filter(([cat, spent]) => {
      const limit = budget.categories[cat];
      return limit && (spent / limit) > 0.9;
    })
    .map(([cat, spent]) => ({
      category: cat,
      ratio: Math.round(spent / budget.categories[cat] * 100),
    }));

  renderWarnings(warnings);
  renderCategoryChart(byCategory);
}
